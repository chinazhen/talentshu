package com.baofoo.example.diamond;

import lombok.extern.slf4j.Slf4j;
import org.apache.commons.lang.StringUtils;
import org.csource.common.NameValuePair;
import org.csource.fastdfs.*;
import javax.annotation.PostConstruct;
import java.io.File;
import java.io.FileInputStream;
import java.io.IOException;
import java.net.InetSocketAddress;
import java.util.HashMap;
import java.util.Map;

/**
 * <ul>
 * <li>FastDfs 测试类</li>
 * <li>User: MuZhi Date:2015/10/22 </li>
 * </ul>
 */
@Slf4j
public class FDfsTest {

    private static boolean status=false;

    public static void main(String args[]){

        FDfsTest FDfsTest = new FDfsTest();

        FDfsTest.initDFS();

        File uploadFile = new File("D:\\mysql.sql");

        NameValuePair[] meta_list = new NameValuePair[]{new NameValuePair("author","shuzhen")};

        try{

            //上传
            Map<String, String> resultMap = FDfsTest.uploadFile(uploadFile, meta_list);
            for(String key:resultMap.keySet()){
                log.info(resultMap.get(key));
            }

            //下载
            int resultCode = FDfsTest.downloadFile("group1","M00/00/00/CgAVOVYoQnCACerzAAAIe8oQSSA496.sql","F://test.sql");
            log.info("resultCode = {}",resultCode);

        }catch (Exception e){
            log.error(e.getMessage());
        }
    }

    @PostConstruct
    public void initDFS(){

        if(status){
            return;
        }

        InetSocketAddress[] tracker_servers = new InetSocketAddress[2];
        tracker_servers[0] = new InetSocketAddress("10.0.21.56", 22122);
        tracker_servers[1] = new InetSocketAddress("10.0.21.57", 22122);
        TrackerGroup trackerGroup = new TrackerGroup(tracker_servers);

        ClientGlobal.setG_tracker_group(trackerGroup);
        ClientGlobal.setG_charset("UTF-8");
        ClientGlobal.setG_secret_key(null);
        ClientGlobal.setG_tracker_http_port(80);
        ClientGlobal.setG_connect_timeout(1000 * 30);
        ClientGlobal.setG_network_timeout(1000 * 60);

    }

    /**
     * 上传文件到FastDFS服务器
     *
     * @param file           file
     * @param meta_list      文件头信息，如宽度，大小，作者等
     * @return               远程返回的文件名称和group名称
     * @throws Exception
     */
    public Map<String, String> uploadFile(File file,NameValuePair[] meta_list) throws Exception {

        initDFS();

        Map<String, String> dfsMap = new HashMap<>();
        FileInputStream fis = null;
        double fileSize = file.length()/1024.0;
        log.info("上传文件大小:{}KB",fileSize);

        TrackerServer trackerServer = null;
        try {
            // 初始化连接
            TrackerClient trackerClient = new TrackerClient();
            trackerServer = trackerClient.getConnection();
            StorageClient storageClient = new StorageClient(trackerServer, null);

            byte[] file_buff = null;
            try{
                // 读取文件
                fis = new FileInputStream(file);
                int len = fis.available();
                file_buff = new byte[len];
                int result = fis.read(file_buff);
                log.debug("fis.read result is :{}",result);

            }catch (IOException e) {
                log.error(e.getMessage(),e);
                throw e;
            }finally{
                try {
                    if (fis != null) {
                        fis.close();
                    }
                } catch (Exception e) {
                    log.error("文件流关闭异常:{}", e);
                }
            }

            String extName = file.getName().substring(file.getName().lastIndexOf(".") + 1);     // 获取后缀名
            String[] results = storageClient.upload_file(file_buff, extName, meta_list);        // 上传文件
            if (results == null) {
                log.error("FastDFS文件上传失败,文件名:{} Error Code:{}",file.getName(), storageClient.getErrorCode());
                throw new Exception("FastDFS文件上传失败");
            }

            log.debug("FastDFS文件上传成功：{} ", file.getAbsolutePath());
            dfsMap.put("GROUP_NAME", results[0]);             // 远程返回的文件名称
            dfsMap.put("REMOTE_FILE_NAME", results[1]);      // 文件的groupId

        } catch (Exception e){
            log.error("FastDFS文件上传异常：{},{}",e.getMessage(),e);
        } finally {
            closeDFSServer(trackerServer);
        }
        return dfsMap;
    }

    /**
     * 下载文件
     *
     * @param groupName         下载组，默认为Group1
     * @param remoteFilename    DFS上面的文件名
     * @param localFileName     下载到本地的文件名
     * @return  int             状态
     */
    public int downloadFile(String groupName, String remoteFilename, String localFileName) {

        initDFS();

        int result = -1;
        TrackerServer trackerServer = null;
        StorageClient storageClient = null;
        try {
            //初始化连接
            TrackerClient trackerClient = new TrackerClient();
            trackerServer = trackerClient.getConnection();
            storageClient = new StorageClient(trackerServer,null);
            //下载文件
            groupName = StringUtils.isEmpty(groupName) ? "group1" : groupName;
            result = storageClient.download_file(groupName, remoteFilename, localFileName);
            log.info("文件下载成功！groupName:{} localFileName{}", groupName, localFileName);
        } catch (Exception e) {
            if(storageClient!=null)
                log.error("文件上传失败！ errorCode:{},{}", storageClient.getErrorCode(), e);
            else
                log.error("文件上传失败！localFileName:{} storageClient return errorCode: null {}", localFileName, e);
        } finally {
            closeDFSServer(trackerServer);
        }
        return result;
    }


    /**
     * 关闭Tracker 连接
     *
     * @param trackerServer trackerServer
     */
    public void closeTrackerServer(TrackerServer trackerServer) {
        // 退出前,一定要将队列服务关闭
        try {
            if (trackerServer != null)
                trackerServer.close();
        } catch (Exception e) {
            log.error("队列服务关闭异常", e);
        }  finally {
            status=false;
        }
    }

    /**
     * 关闭Tracker Storage 连接
     *
     * @param trackerServer 服务追踪者
     */
    public void  closeDFSServer(TrackerServer trackerServer){
        try{
            closeTrackerServer(trackerServer);
        }catch (Exception ioe){
            log.error("DFS关闭连接异常",ioe.getStackTrace());
        } finally {
            status=false;
        }
    }

}
